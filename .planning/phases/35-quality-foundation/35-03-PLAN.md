```python
class RecommendRequest(BaseModel):
    hero_id: int
    role: int  # 1-5
    playstyle: str
    lane_opponents: list[int]  # max 2
    all_opponents: list[int]   # max 5
    allies: list[int]          # max 4
    enemy_context: list[EnemyContext]

class RuleResult(BaseModel):
    item_id: int
    item_name: str
    reasoning: str
    phase: str   # "starting" | "laning" | "core" | "late_game" | "situational"
    priority: str  # "core" | "situational" | "luxury"
    counter_target: str | None = None
```
