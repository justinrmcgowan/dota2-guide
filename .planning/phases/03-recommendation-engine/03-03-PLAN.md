**1. Create `prismlab/backend/engine/recommender.py`** with class `HybridRecommender`:

```python
import time
import logging
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from engine.schemas import (
    RecommendRequest, RecommendResponse, RecommendPhase,
    ItemRecommendation, RuleResult, LLMRecommendation,
)
from engine.rules import RulesEngine
from engine.llm import LLMEngine
from engine.context_builder import ContextBuilder
from data.models import Item

logger = logging.getLogger(__name__)

class HybridRecommender:
    def __init__(self, rules: RulesEngine, llm: LLMEngine, context_builder: ContextBuilder):
        self.rules = rules
        self.llm = llm
        self.context_builder = context_builder

    async def recommend(self, request: RecommendRequest, db: AsyncSession) -> RecommendResponse:
        start = time.monotonic()

        # Step 1: Rules fire instantly (deterministic, no API call)
        rules_items = self.rules.evaluate(request)
        logger.info("Rules engine produced %d recommendations", len(rules_items))

        # Step 2: Build Claude prompt context
        user_message = await self.context_builder.build(request, rules_items, db)

        # Step 3: Call Claude with timeout + fallback
        llm_result = None
        fallback = False
        try:
            llm_result = await self.llm.generate(user_message)
        except Exception as e:
            logger.exception("LLM engine failed: %s", e)
            fallback = True

        if llm_result is None:
            fallback = True

        # Step 4: Merge results
        if llm_result and not fallback:
            phases = self._merge(rules_items, llm_result)
            overall_strategy = llm_result.overall_strategy
        else:
            phases = self._rules_only(rules_items)
            overall_strategy = "Rules-based recommendations only. AI reasoning unavailable."

        # Step 5: Validate all item_ids against DB
        phases = await self._validate_item_ids(phases, db)

        elapsed_ms = int((time.monotonic() - start) * 1000)
        return RecommendResponse(
            phases=phases,
            overall_strategy=overall_strategy,
            fallback=fallback,
            model=LLMEngine.MODEL if not fallback else None,
            latency_ms=elapsed_ms,
        )
```

Implement these private methods:

**`_merge(self, rules_items: list[RuleResult], llm_result: LLMRecommendation) -> list[RecommendPhase]`:**
- Start with LLM phases as the base
- For each RuleResult, find the matching phase in LLM output (by phase name). If phase exists, prepend the rule item to the phase's items list (rules take priority). If phase doesn't exist, create a new RecommendPhase with just the rule item.
- Deduplicate: if a rule item_id already appears in the LLM phase's items, remove the LLM duplicate (keep the rule version since rules are higher priority)
- Convert each RuleResult to ItemRecommendation: `ItemRecommendation(item_id=r.item_id, item_name=r.item_name, reasoning=r.reasoning, priority=r.priority, conditions=None)`

**`_rules_only(self, rules_items: list[RuleResult]) -> list[RecommendPhase]`:**
- Group RuleResult items by phase name
- Return list of RecommendPhase objects, one per group
- Each phase has the grouped items converted to ItemRecommendation

**`async def _validate_item_ids(self, phases: list[RecommendPhase], db: AsyncSession) -> list[RecommendPhase]`:**
- Query all item IDs from the Item table: `SELECT id FROM items`
- Collect into a set of valid IDs
- For each phase, filter out items whose item_id is NOT in the valid set
- Log a warning for each filtered-out item: "Filtered invalid item_id {id} ({name}) from recommendations"
- Return the filtered phases (remove empty phases too)

**1. Create `prismlab/backend/api/routes/recommend.py`:**

```python
import logging
from fastapi import APIRouter, Depends
from sqlalchemy.ext.asyncio import AsyncSession

from data.database import get_db
from data.opendota_client import OpenDotaClient
from engine.schemas import RecommendRequest, RecommendResponse
from engine.rules import RulesEngine
from engine.llm import LLMEngine
from engine.context_builder import ContextBuilder
from engine.recommender import HybridRecommender
from config import settings

logger = logging.getLogger(__name__)
router = APIRouter()

# Singleton instances (created once, reused across requests)
_opendota = OpenDotaClient(api_key=settings.opendota_api_key)
_rules = RulesEngine()
_llm = LLMEngine()
_context_builder = ContextBuilder(opendota_client=_opendota)
_recommender = HybridRecommender(rules=_rules, llm=_llm, context_builder=_context_builder)


@router.post("/recommend", response_model=RecommendResponse)
async def recommend(request: RecommendRequest, db: AsyncSession = Depends(get_db)):
    """Generate item recommendations for a Dota 2 draft context.

    Runs the hybrid engine: rules first (instant), then Claude API (structured output),
    merges results, validates item IDs, and returns phased timeline with reasoning.

    Falls back to rules-only if Claude API fails or times out (10s hard timeout).
    Response includes fallback flag, model name, and latency_ms metadata.
    """
    logger.info(
        "Recommend request: hero=%d role=%d opponents=%s",
        request.hero_id, request.role, request.lane_opponents,
    )
    response = await _recommender.recommend(request, db)
    logger.info(
        "Recommend response: %d phases, fallback=%s, latency=%dms",
        len(response.phases), response.fallback, response.latency_ms or 0,
    )
    return response
```
