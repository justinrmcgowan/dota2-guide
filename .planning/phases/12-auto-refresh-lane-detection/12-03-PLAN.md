    This is the core action that fires a recommendation refresh:
    1. Call `recommend()` -- use the same pattern as ReEvaluateButton. Read stores directly: `useRecommendationStore.getState()` and `useGameStore.getState()` to avoid stale closures. Replicate the `recommend()` logic from `useRecommendation.ts` but call it directly via store access (do NOT import the hook -- hooks can only be called in components). Instead, extract the recommend logic into a standalone async function within useAutoRefresh that:
       - Reads `useGameStore.getState()` for hero, role, etc.
       - Reads `useRecommendationStore.getState()` for purchased items
       - Guards: if no selectedHero or role, return early
       - Guards: if isLoading, return early
       - Calls `store.clearResults()`, `store.setLoading(true)`
       - Builds RecommendRequest (same as useRecommendation.ts)
       - Calls `api.recommend(request)` (import from `../api/client`)
       - Calls `store.setData(response)` or `store.setError(error)`
    2. Set cooldown: `cooldownEndRef.current = Date.now() + 120_000`
    3. Call `useRefreshStore.getState().startCooldown()`
    4. Reset gold baseline: `prevStateRef.current.netWorthAtLastRefresh = currentNetWorth`
    5. Update prev state refs (deaths, roshanState, towers) to current values
    6. Show toast: `useRefreshStore.getState().showToast(event.message)`
    7. Clear queue: `queuedEventRef.current = null; useRefreshStore.getState().clearQueue()`

    **Main useEffect with gsiStore subscription:**

    ```typescript
    useEffect(() => {
      const unsubscribe = useGsiStore.subscribe((state) => {
        // Guard: only process when GSI is connected
        if (state.gsiStatus !== "connected") return;
        const live = state.liveState;
        if (!live) return;

        // Only detect during active game
        if (live.game_state !== "DOTA_GAMERULES_STATE_GAME_IN_PROGRESS") return;

        // Don't auto-refresh if no recommendations exist yet (user hasn't done initial recommend)
        if (!useRecommendationStore.getState().data) return;

        // Don't auto-refresh while a recommendation is loading
        if (useRecommendationStore.getState().isLoading) return;

        // --- Lane auto-detection at 10:00 (D-12, D-13, D-14, GSI-05) ---
        if (
          live.game_clock >= 600 &&
          !laneAutoDetectedRef.current &&
          !laneOverriddenRef.current
        ) {
          const role = useGameStore.getState().role;
          if (role !== null) {
            const result = detectLaneResult(live.gpm, role);
            useGameStore.getState().setLaneResult(result);
            laneAutoDetectedRef.current = true;
            // The 10:00 phase transition trigger will handle the re-evaluation
          }
        }

        // --- Event detection (D-01, D-06) ---
        const currentState = {
          deaths: live.deaths,
          net_worth: live.net_worth,
          roshan_state: live.roshan_state,
          radiant_tower_count: live.radiant_tower_count,
          dire_tower_count: live.dire_tower_count,
          game_clock: live.game_clock,
        };

        const event = detectTriggers(
          currentState,
          prevStateRef.current,
          firedPhasesRef.current,
        );

        if (!event) {
          // Still update non-trigger prev state (deaths, roshan, towers) so we don't miss changes
          // But NOT netWorthAtLastRefresh -- that only resets on refresh (D-04)
          prevStateRef.current.deaths = live.deaths;
          prevStateRef.current.roshanState = live.roshan_state;
          prevStateRef.current.radiantTowers = live.radiant_tower_count;
          prevStateRef.current.direTowers = live.dire_tower_count;
          return;
        }

        // --- Cooldown check (D-07, D-08) ---
        const now = Date.now();
        if (now < cooldownEndRef.current) {
          // Queue the event (latest replaces previous per D-08)
          queuedEventRef.current = event;
          useRefreshStore.getState().queueEvent(event);
          // Update prev state for non-gold fields to avoid re-detection
          prevStateRef.current.deaths = live.deaths;
          prevStateRef.current.roshanState = live.roshan_state;
          prevStateRef.current.radiantTowers = live.radiant_tower_count;
          prevStateRef.current.direTowers = live.dire_tower_count;
          return;
        }

        // Fire immediately
        fireRefresh(event, live.net_worth);
      });

      // --- Cooldown expiry check (1Hz interval) (D-08, D-10) ---
      const interval = setInterval(() => {
        const now = Date.now();

        // Update countdown display (D-10)
        useRefreshStore.getState().tick(now);

        // Check if queued event should fire
        const queued = queuedEventRef.current;
        if (queued && now >= cooldownEndRef.current) {
          queuedEventRef.current = null;
          useRefreshStore.getState().clearQueue();

          // Get current net worth for baseline reset
          const live = useGsiStore.getState().liveState;
          const nw = live?.net_worth ?? prevStateRef.current.netWorthAtLastRefresh;
          fireRefresh(queued, nw);
        }
      }, 1000);

      return () => {
        unsubscribe();
        clearInterval(interval);
      };
    }, []);
    ```

    ```typescript
    import { useState, useEffect } from "react";
    import { useRefreshStore } from "../../stores/refreshStore";

    function AutoRefreshToast() {
      const toast = useRefreshStore((s) => s.lastToast);
      const dismiss = useRefreshStore((s) => s.dismissToast);
      const [visible, setVisible] = useState(false);

      useEffect(() => {
        if (!toast) {
          setVisible(false);
          return;
        }
        setVisible(true);
        const timer = setTimeout(() => {
          setVisible(false);
          // Small delay to allow exit animation before clearing state
          setTimeout(() => dismiss(), 300);
        }, 4000);
        return () => clearTimeout(timer);
      }, [toast?.timestamp, dismiss]);

      if (!toast) return null;

      return (
        <div
          className={`fixed bottom-4 right-4 z-50 flex items-start gap-3 rounded-lg border border-bg-elevated bg-bg-secondary p-4 shadow-xl transition-all duration-300 ${
            visible
              ? "opacity-100 translate-y-0"
              : "opacity-0 translate-y-2 pointer-events-none"
          }`}
        >
          {/* Lightning bolt SVG icon in cyan */}
          <svg
            width="20"
            height="20"
            viewBox="0 0 24 24"
            fill="none"
            stroke="currentColor"
            strokeWidth={2}
            strokeLinecap="round"
            strokeLinejoin="round"
    ```

       ```tsx
          {secondsRemaining > 0 && queuedEvent && (
       ```
