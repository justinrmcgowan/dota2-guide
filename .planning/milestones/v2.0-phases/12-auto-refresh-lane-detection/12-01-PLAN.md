    **Step 1: Create laneBenchmarks.ts**

    File: `prismlab/frontend/src/utils/laneBenchmarks.ts`

    Per D-11, export:
    ```typescript
    export const GPM_BENCHMARKS: Record<number, number> = {
      1: 500,  // Pos 1 Carry
      2: 480,  // Pos 2 Mid
      3: 400,  // Pos 3 Off
      4: 280,  // Pos 4 Soft Sup
      5: 230,  // Pos 5 Hard Sup
    };

    export function detectLaneResult(
      gpm: number,
      role: number,
    ): "won" | "even" | "lost" {
      const benchmark = GPM_BENCHMARKS[role];
      if (!benchmark) return "even"; // Unknown role, safe default

      if (gpm > benchmark * 1.10) return "won";
      if (gpm < benchmark * 0.90) return "lost";
      return "even";
    }
    ```

    **Step 2: Create triggerDetection.ts**

    File: `prismlab/frontend/src/utils/triggerDetection.ts`

    Per D-01, D-02, D-03, D-04, D-05, D-06:

    Export types:
    ```typescript
    export interface TriggerEvent {
      type: "death" | "gold_swing" | "tower_kill" | "roshan_kill" | "phase_transition";
      message: string;
    }

    export interface PreviousState {
      deaths: number;
      netWorthAtLastRefresh: number;
      roshanState: string;
      radiantTowers: number;
      direTowers: number;
    }

    export interface CurrentState {
      deaths: number;
      net_worth: number;
      roshan_state: string;
      radiant_tower_count: number;
      dire_tower_count: number;
      game_clock: number;
    }
    ```

    Export function `detectTriggers(current: CurrentState, prev: PreviousState, firedPhases: Set<number>): TriggerEvent | null`

    Priority order (per D-16 message text):
    1. Phase transitions (check 2100 first, then 1200, then 600 -- highest threshold first to catch if multiple crossed at once):
       - clock >= 2100 && !firedPhases.has(2100): add to set, return `{ type: "phase_transition", message: "Late game reached (35:00)" }`
       - clock >= 1200 && !firedPhases.has(1200): add to set, return `{ type: "phase_transition", message: "Mid-game reached (20:00)" }`
       - clock >= 600 && !firedPhases.has(600): add to set, return `{ type: "phase_transition", message: "Lane phase ended (10:00)" }`
    2. Death: current.deaths > prev.deaths: return `{ type: "death", message: "Death -- reassessing priorities" }`
    3. Roshan: current.roshan_state !== prev.roshanState && current.roshan_state === "respawn_base": return `{ type: "roshan_kill", message: "Roshan killed -- updating" }`
    4. Tower kill: total towers decreased: `(current.radiant_tower_count + current.dire_tower_count) < (prev.radiantTowers + prev.direTowers)`: return `{ type: "tower_kill", message: "Tower destroyed -- map changed" }`
    5. Gold swing: `Math.abs(current.net_worth - prev.netWorthAtLastRefresh) >= 2000`:
       - Compute delta = current.net_worth - prev.netWorthAtLastRefresh
       - Format: `const sign = delta > 0 ? "+" : ""; return { type: "gold_swing", message: \`Gold swing: ${sign}${Math.abs(delta).toLocaleString()}g\` }`
    6. Return null if nothing detected.

    Note: Item purchase is NOT a trigger (per D-02).
