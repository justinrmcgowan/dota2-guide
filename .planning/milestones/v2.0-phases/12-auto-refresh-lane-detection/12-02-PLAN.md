    **Create refreshStore.ts**

    File: `prismlab/frontend/src/stores/refreshStore.ts`

    Per D-07, D-08, D-09, D-10, D-15, D-17:

    ```typescript
    import { create } from "zustand";

    export interface TriggerEvent {
      type: "death" | "gold_swing" | "tower_kill" | "roshan_kill" | "phase_transition";
      message: string;
    }

    interface RefreshStore {
      // Cooldown state (D-07, D-08)
      cooldownEnd: number | null;       // timestamp ms when cooldown expires
      queuedEvent: TriggerEvent | null; // most recent event during cooldown (D-08)
      secondsRemaining: number;         // countdown display value for UI (D-10)

      // Toast state (D-15, D-16, D-17)
      lastToast: { message: string; timestamp: number } | null;

      // Actions
      startCooldown: () => void;        // Sets 120s cooldown from now (D-07)
      queueEvent: (event: TriggerEvent) => void;  // Queue-latest-only (D-08)
      clearQueue: () => void;
      showToast: (message: string) => void;   // D-15
      dismissToast: () => void;
      tick: (now: number) => void;      // Update secondsRemaining for countdown display (D-10)
      resetCooldown: () => void;        // Full reset (for game end / disconnect)
    }
    ```

    Implementation:
    - `startCooldown`: `set({ cooldownEnd: Date.now() + 120_000 })`
    - `queueEvent`: `set({ queuedEvent: event })` -- always replaces, per D-08
    - `clearQueue`: `set({ queuedEvent: null })`
    - `showToast`: `set({ lastToast: { message, timestamp: Date.now() } })`
    - `dismissToast`: `set({ lastToast: null })`
    - `tick(now)`: If `cooldownEnd` is null or `now >= cooldownEnd`, set `secondsRemaining: 0`. Otherwise, set `secondsRemaining: Math.ceil((cooldownEnd - now) / 1000)`.
    - `resetCooldown`: `set({ cooldownEnd: null, queuedEvent: null, secondsRemaining: 0 })`

    Note: The `TriggerEvent` type is re-exported here for convenience, but the canonical definition is in `utils/triggerDetection.ts` (Plan 01). If Plan 01 runs first, import from there. If Plan 02 runs first (parallel wave), define locally -- Plan 03 will reconcile to import from triggerDetection.ts.
