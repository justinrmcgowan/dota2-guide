**prismlab/frontend/src/utils/heroSearch.ts** -- Fuse.js search wrapper:
```typescript
import Fuse from "fuse.js";
import type { Hero } from "../types/hero";

const fuseOptions: Fuse.IFuseOptions<Hero> = {
  keys: [
    { name: "localized_name", weight: 2 },
    { name: "roles", weight: 0.5 },
  ],
  threshold: 0.4,
  distance: 100,
  minMatchCharLength: 2,
};

export function createHeroSearcher(heroes: Hero[]): Fuse<Hero> {
  return new Fuse(heroes, fuseOptions);
}

export function searchHeroes(fuse: Fuse<Hero>, query: string): Hero[] {
  if (!query.trim()) return [];
  return fuse.search(query).map((result) => result.item);
}
```
- localized_name has weight 2 (highest priority for name matching)
- roles has weight 0.5 (secondary search by role like "Carry" or "Support")
- threshold 0.4 catches typos like "anit" for "Anti-Mage" and abbreviations like "jug" for "Juggernaut"
- minMatchCharLength 2 prevents single-character noise

**prismlab/frontend/src/hooks/useHeroes.ts** -- fetch and cache hero list:
```typescript
import { useState, useEffect } from "react";
import type { Hero } from "../types/hero";
import { api } from "../api/client";

export function useHeroes() {
  const [heroes, setHeroes] = useState<Hero[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    let cancelled = false;
    api.getHeroes()
      .then((data) => {
        if (!cancelled) {
          setHeroes(data);
          setLoading(false);
        }
      })
      .catch((err) => {
        if (!cancelled) {
          setError(err.message);
          setLoading(false);
        }
      });
    return () => { cancelled = true; };
  }, []);

  return { heroes, loading, error };
}
```
- Fetches once on mount, caches in state
- Cleanup prevents state updates on unmounted component
- Returns loading/error states for UI feedback
